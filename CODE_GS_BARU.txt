function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const action = e && e.parameter ? String(e.parameter.action || "") : "";

  if (action === "getOptions") {
    const tipe = String(e.parameter.tipe || "");
    return ContentService
      .createTextOutput(JSON.stringify(buildOptionData(ss, tipe)))
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (action === "ping") {
    return ContentService
      .createTextOutput(JSON.stringify({
        success: true,
        message: "Apps Script aktif",
        sheets: listAvailableSheets(ss)
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService
    .createTextOutput(JSON.stringify({
      success: true,
      master: [],
      proyek: []
    }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const transaksiSheet = requireSheet(ss, SHEET_TRANSAKSI);
    const logSheet = requireSheet(ss, SHEET_LOG);
    const adminSheet = requireSheet(ss, SHEET_ADMIN);
    const payload = JSON.parse(e.postData.contents || "{}");
    const settings = readAdminSettings(adminSheet);

    validatePayload(ss, payload, settings);

    const id = "TRX-" + new Date().getTime();
    const timestamp = new Date();
    const nominal = sanitizeNumber(payload.nominal);
    const imageUrl = processProofUpload(payload, settings, id);

    const tanggalValue = parseDateInput_(payload.tanggal);

    transaksiSheet.appendRow([
      id,
      tanggalValue,
      payload.proyek || "",
      payload.kategori || "",
      payload.deskripsi || "",
      nominal,
      payload.metode || "",
      payload.pic || "",
      payload.catatan || "",
      imageUrl,
      timestamp,
      payload.user_input || ""
    ]);

    logSheet.appendRow([
      timestamp,
      "Tambah Transaksi",
      payload.user_input || "Sistem",
      [id, formatDateForLog_(tanggalValue), payload.proyek || "", payload.kategori || "", nominal].join(" | ")
    ]);

    refreshDashboardSheet();

    return ContentService
      .createTextOutput(JSON.stringify({
        success: true,
        status: "success",
        id: id,
        timestamp: timestamp.toISOString(),
        message: "Transaksi berhasil disimpan ke sheet Transaksi"
      }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({
        success: false,
        status: "error",
        error: error.message
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function buildOptionData(ss, tipe) {
  const masterSheet = requireSheet(ss, SHEET_MASTER);
  const proyekSheet = requireSheet(ss, SHEET_PROYEK);
  const masterRows = masterSheet.getDataRange().getValues().slice(1).filter(notEmptyRow);
  const proyekRows = proyekSheet.getDataRange().getValues().slice(1).filter(notEmptyRow);

  const masterData = masterRows.map(function (row) {
    return {
      tipe: String(row[0] || ""),
      id: String(row[1] || ""),
      nama: String(row[2] || ""),
      status: String(row[3] || ""),
      urutan: Number(row[4] || 0)
    };
  });

  const proyekData = proyekRows.map(function (row) {
    return {
      id: String(row[0] || ""),
      nama_proyek: String(row[1] || ""),
      klien: String(row[2] || ""),
      lokasi: String(row[3] || ""),
      status: String(row[4] || "")
    };
  });

  if (!tipe) {
    return { master: masterData, proyek: proyekData };
  }

  return {
    master: masterData
      .filter(function (item) {
        return normalize(item.tipe) === normalize(tipe) && normalize(item.status) === "aktif";
      })
      .sort(function (a, b) {
        return a.urutan - b.urutan;
      }),
    proyek: proyekData.filter(function (item) {
      return normalize(item.status) === "aktif";
    })
  };
}

function validatePayload(ss, payload, settings) {
  const requiredFields = ["tanggal", "proyek", "kategori", "nominal", "metode", "pic", "deskripsi", "user_input"];

  requiredFields.forEach(function (field) {
    if (!String(payload[field] || "").trim()) {
      throw new Error("Field wajib kosong: " + field);
    }
  });

  const nominal = sanitizeNumber(payload.nominal);
  if (nominal <= 0) {
    throw new Error("Nominal harus lebih dari 0");
  }

  if (settings.maxNominal > 0 && nominal > settings.maxNominal) {
    throw new Error("Nominal melebihi batas maksimal di sheet Admin");
  }

  if (!isActiveProject(ss, payload.proyek)) {
    throw new Error("Proyek tidak ditemukan atau statusnya tidak aktif");
  }

  ["Kategori", "Metode", "PIC"].forEach(function (tipe) {
    const key = tipe === "Kategori" ? "kategori" : tipe === "Metode" ? "metode" : "pic";
    if (!isActiveMasterValue(ss, tipe, payload[key])) {
      throw new Error(tipe + " tidak ditemukan atau statusnya tidak aktif");
    }
  });
}

function processProofUpload(payload, settings, id) {
  if (normalize(settings.uploadNota) !== "true") {
    return "Upload dinonaktifkan dari sheet Admin";
  }

  if (!(payload.bukti && typeof payload.bukti === "string" && payload.bukti.indexOf("base64") !== -1)) {
    return "";
  }

  try {
    const folder = FOLDER_ID ? DriveApp.getFolderById(FOLDER_ID) : DriveApp.getRootFolder();
    const splitBase = payload.bukti.split(",");
    const dataStr = splitBase[1] || payload.bukti;
    const contentType = splitBase[0].split(";")[0].replace("data:", "") || "image/jpeg";
    const fileExtension = guessExtension(contentType);
    const filename = ["Bukti", sanitizeFileName(payload.proyek || "ADK"), id].join("_") + "." + fileExtension;
    const blob = Utilities.newBlob(Utilities.base64Decode(dataStr), contentType, filename);
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return file.getUrl();
  } catch (uploadError) {
    return "Upload File Gagal: " + uploadError.message;
  }
}

function readAdminSettings(sheet) {
  const defaults = {
    companyName: "CV Akbar Dharma Karya",
    currency: "IDR",
    dateFormat: "dd/MM/yyyy",
    maxNominal: 0,
    uploadNota: "TRUE"
  };

  const rows = sheet.getDataRange().getValues().slice(1).filter(notEmptyRow);
  rows.forEach(function (row) {
    const key = String(row[0] || "");
    const value = row[1];

    if (key === "NamaPerusahaan") defaults.companyName = String(value || defaults.companyName);
    if (key === "MataUang") defaults.currency = String(value || defaults.currency);
    if (key === "FormatTanggal") defaults.dateFormat = String(value || defaults.dateFormat);
    if (key === "MaksNominal") defaults.maxNominal = sanitizeNumber(value);
    if (key === "UploadNota") defaults.uploadNota = String(value || defaults.uploadNota);
  });

  return defaults;
}

function isActiveProject(ss, projectName) {
  const rows = requireSheet(ss, SHEET_PROYEK).getDataRange().getValues().slice(1).filter(notEmptyRow);
  return rows.some(function (row) {
    return normalize(row[1]) === normalize(projectName) && normalize(row[4]) === "aktif";
  });
}

function isActiveMasterValue(ss, tipe, value) {
  const rows = requireSheet(ss, SHEET_MASTER).getDataRange().getValues().slice(1).filter(notEmptyRow);
  return rows.some(function (row) {
    return normalize(row[0]) === normalize(tipe) && normalize(row[2]) === normalize(value) && normalize(row[3]) === "aktif";
  });
}

function buildDashboardLayout(sheet) {
  // Hapus semua isi lama termasuk formula QUERY yang error
  sheet.clear();

  // Hapus semua chart lama
  var oldCharts = sheet.getCharts();
  for (var ci = 0; ci < oldCharts.length; ci++) {
    sheet.removeChart(oldCharts[ci]);
  }

  sheet.setHiddenGridlines(true);

  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var transaksiSheet = ss.getSheetByName(SHEET_TRANSAKSI);
  var proyekSheet = ss.getSheetByName(SHEET_PROYEK);

  if (!transaksiSheet || !proyekSheet) {
    sheet.getRange("A1").setValue("Dashboard belum siap. Jalankan 'Setup Semua Sheet' dari menu CVADK Tools.");
    return;
  }

  var transactions = readTransactions_(transaksiSheet);
  var proyekAktif = countActiveProjects_(proyekSheet);
  var today = new Date();
  var todayKey = Utilities.formatDate(today, Session.getScriptTimeZone(), "yyyy-MM-dd");
  var lastUpdate = Utilities.formatDate(today, Session.getScriptTimeZone(), "dd MMM yyyy, HH:mm");

  var totalTransaksi = transactions.length;
  var totalNominal = 0;
  var transaksiHariIni = [];
  var totalHariIni = 0;
  for (var i = 0; i < transactions.length; i++) {
    totalNominal += transactions[i].nominal;
    if (transactions[i].tanggalKey === todayKey) {
      transaksiHariIni.push(transactions[i]);
      totalHariIni += transactions[i].nominal;
    }
  }
  var rataRata = totalTransaksi > 0 ? Math.round(totalNominal / totalTransaksi) : 0;

  var byKategori = summarizeByKey_(transactions, "kategori");
  var byProyek = summarizeByKey_(transactions, "proyek");
  var byMetode = summarizeByKey_(transactions, "metode");
  var recent = transactions.slice()
    .sort(function (a, b) { return b.timestamp.getTime() - a.timestamp.getTime(); })
    .slice(0, 8);

  // === SET COLUMN WIDTHS (untuk layout yang rapi dan konsisten) ===
  sheet.setColumnWidth(1, 30);   // A - spacer kiri
  sheet.setColumnWidth(2, 180);  // B
  sheet.setColumnWidth(3, 160);  // C
  sheet.setColumnWidth(4, 30);   // D - spacer
  sheet.setColumnWidth(5, 180);  // E
  sheet.setColumnWidth(6, 160);  // F
  sheet.setColumnWidth(7, 30);   // G - spacer
  sheet.setColumnWidth(8, 180);  // H
  sheet.setColumnWidth(9, 160);  // I
  sheet.setColumnWidth(10, 30);  // J - spacer
  sheet.setColumnWidth(11, 160); // K
  sheet.setColumnWidth(12, 120); // L
  sheet.setColumnWidth(13, 120); // M

  // === ROW 1-2: HEADER UTAMA ===
  sheet.setRowHeight(1, 55);
  sheet.setRowHeight(2, 30);
  sheet.getRange("A1:M1").merge().setValue("DASHBOARD KEUANGAN")
    .setFontFamily("Google Sans").setFontSize(22).setFontWeight("bold")
    .setHorizontalAlignment("center").setVerticalAlignment("middle")

  sheet.getRange("A2:M2").merge().setValue("CV AKBAR DHARMA KARYA  ·  Terakhir diperbarui: " + lastUpdate)
    .setFontFamily("Google Sans").setFontSize(10)
    .setHorizontalAlignment("center").setVerticalAlignment("middle")
    .setBackground(COLORS.subHeaderBg).setFontColor(COLORS.subHeaderText);

  // === ROW 3: SPACER ===

// ─── KPI CARD BUILDER ───
function buildKpiCard_(sheet, row, col, label, value, format, accentColor) {
  // Label (baris atas)
  sheet.getRange(row, col, 1, 2).merge().setValue(label)
    .setFontFamily("Google Sans").setFontSize(8).setFontColor(COLORS.mutedText)
    .setHorizontalAlignment("center").setVerticalAlignment("bottom")
    .setBackground(COLORS.kpiLabelBg);

  // Value (baris bawah)
  var valueCell = sheet.getRange(row + 1, col, 1, 2).merge().setValue(value)
    .setFontFamily("Google Sans").setFontSize(18).setFontWeight("bold")
    .setFontColor(accentColor).setHorizontalAlignment("center").setVerticalAlignment("middle")
    .setBackground("#ffffff");

  if (format) {
    valueCell.setNumberFormat(format);
  }

  // Border (accent warna di atas card)
  sheet.getRange(row, col, 2, 2).setBorder(true, true, true, true, false, false, accentColor, SpreadsheetApp.BorderStyle.SOLID);
}

function seedAdminSheet(sheet) {
  const existing = sheet.getDataRange().getValues().slice(1).filter(notEmptyRow);
  if (existing.length > 0) {
    return;
  }

  sheet.getRange(2, 1, 5, 3).setValues([
    ["NamaPerusahaan", "CV Akbar Dharma Karya", "Nama perusahaan"],
    ["MataUang", "IDR", "Currency"],
    ["FormatTanggal", "dd/MM/yyyy", "Format tanggal tampilan"],
    ["MaksNominal", 100000000, "Batas nominal transaksi"],
    ["UploadNota", "TRUE", "TRUE untuk aktifkan upload bukti"]
  ]);
}

function formatDatabaseSheets(transaksiSheet, proyekSheet, masterSheet, adminSheet, logSheet) {
  transaksiSheet.getRange("B:B").setNumberFormat("dd/MM/yyyy");
  transaksiSheet.getRange("F:F").setNumberFormat("#,##0");
  transaksiSheet.getRange("K:K").setNumberFormat("dd/MM/yyyy HH:mm:ss");
  proyekSheet.getRange("A:E").setHorizontalAlignment("left");
  masterSheet.getRange("A:E").setHorizontalAlignment("left");
  adminSheet.getRange("A:C").setHorizontalAlignment("left");
  logSheet.getRange("A:D").setHorizontalAlignment("left");
}

function ensureHeaders(sheet, headers) {
  const current = sheet.getRange(1, 1, 1, headers.length).getValues()[0];
  const mismatch = headers.some(function (header, index) {
    return String(current[index] || "") !== header;
  });

  if (mismatch) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");
    sheet.setFrozenRows(1);
  }
}

function sanitizeNumber(value) {
  const clean = String(value || "0").replace(/[^0-9.-]/g, "");
  const parsed = parseFloat(clean);
  return isNaN(parsed) ? 0 : parsed;
}

function normalize(value) {
  return String(value || "").trim().toLowerCase();
}

  return row.join("") !== "";
}

  return String(name || "file").replace(/[\\/:*?\"<>|]/g, "_");
}

function guessExtension(contentType) {
  if (contentType === "image/png") return "png";
  if (contentType === "image/webp") return "webp";
  return "jpg";
}

function readTransactions_(sheet) {
  var rows = sheet.getDataRange().getValues().slice(1).filter(notEmptyRow);
  var result = [];
  for (var i = 0; i < rows.length; i++) {
    var row = rows[i];
    var tanggal = toDate_(row[1]);
    var timestamp = toDate_(row[10]);
    // Truncate bukti agar tidak melebihi batas 50K karakter per sel
    var buktiRaw = String(row[9] || "");
    var buktiSafe = buktiRaw;
    if (buktiRaw.length > 200) {
      // Jika ini URL (link Drive), simpan utuh. Jika base64, potong.
      if (buktiRaw.indexOf("http") === 0) {
        buktiSafe = buktiRaw;
      } else {
        buktiSafe = buktiRaw.substring(0, 100) + "... (data terpotong)";
      }
    }
    result.push({
      id: String(row[0] || ""),
      tanggal: tanggal,
      tanggalKey: tanggal ? Utilities.formatDate(tanggal, Session.getScriptTimeZone(), "yyyy-MM-dd") : "",
      proyek: String(row[2] || ""),
      kategori: String(row[3] || ""),
      deskripsi: String(row[4] || ""),
      nominal: sanitizeNumber(row[5]),
      metode: String(row[6] || ""),
      pic: String(row[7] || ""),
      catatan: String(row[8] || ""),
      bukti: buktiSafe,
      timestamp: timestamp || new Date(0),
      user: String(row[11] || "")
    });
  }
  return result;
}

function toDate_(value) {
  if (Object.prototype.toString.call(value) === "[object Date]" && !isNaN(value.getTime())) {
    return value;
  }
  if (!value) return null;
  const parsed = new Date(value);
  if (!isNaN(parsed.getTime())) return parsed;
  return null;
}

function countActiveProjects_(sheet) {
  const rows = sheet.getDataRange().getValues().slice(1).filter(notEmptyRow);
  return rows.filter(function (row) {
    return normalize(row[4]) === "aktif";
  }).length;
}

function summarizeByKey_(rows, key) {
  const map = {};
  rows.forEach(function (row) {
    const label = String(row[key] || "").trim() || "Lainnya";
    if (!map[label]) {
      map[label] = { label: label, total: 0 };
    }
    map[label].total += row.nominal;
  });

  return Object.keys(map)
    .map(function (k) { return map[k]; })
    .sort(function (a, b) { return b.total - a.total; })
    .slice(0, 10);
}

function writeSummaryTable_(sheet, startRow, startCol, rows, labelHeader) {
  // Header tabel
  sheet.getRange(startRow, startCol, 1, 2).setValues([[labelHeader, "Total (Rp)"]])
    .setFontFamily("Google Sans").setFontSize(10).setFontWeight("bold")
    .setBackground(COLORS.tableHeaderBg).setFontColor(COLORS.tableHeaderText)
    .setHorizontalAlignment("center");

  if (!rows.length) {
    sheet.getRange(startRow + 1, startCol, 1, 2).setValues([["Belum ada data", 0]])
      .setFontFamily("Google Sans").setFontSize(9).setFontColor(COLORS.mutedText);
    return;
  }

  var values = [];
  for (var i = 0; i < rows.length; i++) {
    values.push([rows[i].label, rows[i].total]);
  }
  var dataRange = sheet.getRange(startRow + 1, startCol, values.length, 2);
  dataRange.setValues(values).setFontFamily("Google Sans").setFontSize(9);
  sheet.getRange(startRow + 1, startCol + 1, values.length, 1).setNumberFormat("#,##0");

  // Alternating row colors
  for (var r = 0; r < values.length; r++) {
    var bgColor = (r % 2 === 0) ? COLORS.tableRowEven : COLORS.tableRowOdd;
    sheet.getRange(startRow + 1 + r, startCol, 1, 2).setBackground(bgColor);
  }

  // Border seluruh tabel
  sheet.getRange(startRow, startCol, values.length + 1, 2)
    .setBorder(true, true, true, true, true, true, COLORS.tableBorder, SpreadsheetApp.BorderStyle.SOLID);
}

function parseDateInput_(value) {
  const raw = String(value || "").trim();
  if (!raw) {
    throw new Error("Tanggal wajib diisi");
  }

  // Format input date browser biasanya yyyy-MM-dd
  if (/^\d{4}-\d{2}-\d{2}$/.test(raw)) {
    const parts = raw.split("-");
    const year = Number(parts[0]);
    const month = Number(parts[1]) - 1;
    const day = Number(parts[2]);
    return new Date(year, month, day);
  }

  // Dukungan jika user memberi dd/MM/yyyy
  if (/^\d{2}\/\d{2}\/\d{4}$/.test(raw)) {
    const parts = raw.split("/");
    const day = Number(parts[0]);
    const month = Number(parts[1]) - 1;
    const year = Number(parts[2]);
    return new Date(year, month, day);
  }

  const parsed = new Date(raw);
  if (!isNaN(parsed.getTime())) {
    return parsed;
  }

  throw new Error("Format tanggal tidak valid. Gunakan tanggal yang benar.");
}

function formatDateForLog_(value) {
  if (Object.prototype.toString.call(value) === "[object Date]" && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), "dd/MM/yyyy");
  }
  return String(value || "");
}
